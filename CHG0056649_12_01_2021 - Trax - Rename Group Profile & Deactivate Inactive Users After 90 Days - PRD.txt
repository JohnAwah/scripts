# Define functions.

errorcheck()
{
  if [ $? != 0 ]
  then
    echo "Error during LOCK USER ACCOUNTS script execution."
    failure;
    exit 1
  fi
}

failure()
{
  echo " "
  echo aborting $0
  END_DATE=`date +%m/%d/%Y`
  END_TIME=`date +%H:%M:%S`
  if [ ${EMAIL_NOTIFICATION} == TRUE ]
  then
    MAILTO=${LEVEL1_RECIPIENTS}
    MAIL_BODY="Error occurred during execution of ${SCRIPT_NAME} script for ${DB_NAME} database on ${HOST_NAME}"
    EmailSubject="LEVEL1-${CLIENT}-${DB_NAME}-${SCRIPT_NAME} script execution for ${DB_NAME} database on ${HOST_NAME} Failed."
    send_mail;
  fi
  echo "$0 ABORTED on ${END_DATE} ${END_TIME}"
  exit 1
}

success()
{
  echo " "
  END_DATE=`date +%m/%d/%Y`
  END_TIME=`date +%H:%M:%S`
  if [ ${EMAIL_NOTIFICATION} == TRUE ]
  then
    MAILTO=${LEVEL4_RECIPIENTS}
    MAIL_BODY="LOCK USER ACCOUNTS script execution successful for ${DB_NAME} on ${HOST_NAME}. Please check the attached file for the script output."
    EmailSubject="LEVEL4-${CLIENT}-${DB_NAME}-${SCRIPT_NAME} script execution successful for ${DB_NAME} on ${HOST_NAME}.."
    send_mail;
  fi
  echo "$0 SUCCEDED on ${END_DATE} ${END_TIME}"
  exit 0
}
